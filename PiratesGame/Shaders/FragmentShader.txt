in vec3 passed;
in vec3 n;
in vec3 l;
in float distanceSquare;

float hue = 207;

float c = (1 - abs((2 * LI) - 1));
float x = c * (1 - abs(mod((hue/60), 2) - 1));
float m = LI - c/1;

void main() {
	
	vec3 intermColor;

	int test = int(hue/60);

	switch(test){

			intermColor = vec3(c, x, 0);
			break;
		case 1:
			intermColor = vec3(x, c, 0);
			break;
		case 2:
			intermColor = vec3(0, c, x);
			break;
		case 3:
